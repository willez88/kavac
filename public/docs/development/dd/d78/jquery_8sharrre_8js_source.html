453  /* getSocialJson methode
454  ================================================== */
455  Plugin.prototype.getSocialJson = function(name) {
456  var self = this,
457  count = 0,
458  url = urlJson[name].replace('{url}', encodeURIComponent(this.options.url));
459  if (this.options.buttons[name].urlCount === true && this.options.buttons[name].url !== '') {
460  url = urlJson[name].replace('{url}', this.options.buttons[name].url);
461  }
462  //console.log('name : ' + name + ' - url : '+url); //debug
463  if (url != '' && self.options.urlCurl !== '') { //urlCurl = '' if you don't want to used PHP script but used social button
464  $.getJSON(url, function(json) {
465  if (typeof json.count !== "undefined") { //GooglePlus, Stumbleupon, Twitter, Pinterest and Digg
466  var temp = json.count + '';
467  temp = temp.replace('\u00c2\u00a0', ''); //remove google plus special chars
468  count += parseInt(temp, 10);
469  }
470  //get the FB total count (shares, likes and more)
471  else if (json.data && json.data.length > 0 && typeof json.data[0].total_count !== "undefined") { //Facebook total count
472  count += parseInt(json.data[0].total_count, 10);
473  } else if (typeof json[0] !== "undefined") { //Delicious
474  count += parseInt(json[0].total_posts, 10);
475  } else if (typeof json[0] !== "undefined") { //Stumbleupon
476  }
477  self.options.count[name] = count;
478  self.options.total += count;
479  self.renderer();
480  self.rendererPerso();
481  //console.log(json); //debug
482  })
483  .error(function() {
484  self.options.count[name] = 0;
485  self.rendererPerso();
486  });
487  } else {
488  self.renderer();
489  self.options.count[name] = 0;
490  self.rendererPerso();
491  }
492  };

504  }

523  '</div>'
524  );
525  }

528  /* format total numbers like 1.2k or 5M
529  ================================================== */
530  Plugin.prototype.shorterTotal = function(num) {
