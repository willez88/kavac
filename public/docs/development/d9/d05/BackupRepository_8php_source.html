77  *

81  public function restore($filename, $request)
82  {
83  /** @var string Ruta en la que se realiza el respaldo de la base de datos */
84  $path = Storage::disk(config('backup.backup.destination.disks')[0])->getAdapter()->getPathPrefix();
85  /** @var ZipArchive Instancia a la clase */
86  $zip = new ZipArchive;
87  /** @var object Objeto con información del archivo a descomprimir */
88  $res = $zip->open($path . config('app.name') . "/" . $filename);
89  if ($res === true) {
90  try {
91  $request->session()->regenerate();
92  /** @var string Nombre del archivo a descomprimir */
93  $snapName = str_replace(".zip", "", $filename);
94  $zip->renameName('db-dumps/postgresql-kavac.sql', $snapName . '.sql');
95  $zip->extractTo($path);
96  $zip->close();
97  /** @var string Código de respuesta en la restauración de datos */
98  $exitCode = Artisan::call('snapshot:load ' . $snapName);
99  /** @var string Datos de respuesta en la ejecución del comando */
100  $output = Artisan::output();
101  /** log the results */
102  Log::info(
103  "Backpack\BackupManager -- " .
104  'se ha generado una nueva restauración de la base de datos desde la interfaz administrativa' .
105  ". \r\n" .
106  $output
107  );
108  return (strpos($output, 'loaded') > 0);
109  } catch (Exception $e) {
110  return false;
111  }
112  }
113 
114  return false;

129  public function getList($disk, $dir)
130  {
131  /** @var object Objeto con información del disco a usar para obtener el listado de archivos disponibles */
132  $storage_disk = Storage::disk($disk[0]);
133  /** @var object Objeto con información de los archivos disponibles */
134  $files = $storage_disk->files($dir);
135  /** @var array Listado con información detallada de los archivos a restaurar */
136  $backups = [];
137  /** make an array of backup files, with their filesize and creation date */
138  foreach ($files as $k => $f) {
139  /** only take the zip files into account */
140  if (substr($f, -4) == '.zip' && $storage_disk->exists($f)) {
141  $backups[] = [
142  'file_path' => $f,
143  'file_name' => str_replace($dir . '/', '', $f),
144  'file_size' => $this->humanFileSize($storage_disk->size($f)),
145  'last_modified' => $storage_disk->lastModified($f),
146  ];
147  }
148  }
149  /** reverse the backups, so the newest one would be on top */
150  return array_reverse($backups);
151  }
