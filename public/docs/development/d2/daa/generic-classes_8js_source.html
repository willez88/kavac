234  * Gestiona y registra los errores generados en la interfaz de usuario

236  * @author Ing. Roldan Vargas <rvargas@cenditel.gob.ve> | <roldandvg@gmail.com>
237  * @license <a href='http://conocimientolibre.cenditel.gob.ve/licencia-de-software-v-1-3/'>LICENCIA DE SOFTWARE CENDITEL</a>
238  *
239  * Ejemplo: throw new HandleJSError('app', {
240  * log_url: configuración de la url que genera los logs de la interfaz del sistema,
241  * debug: Determina si la aplicación se encuentra o no en modo debug
242  * message: Mensaje de error generado
243  });
244  */
245 class HandleJSError extends Error {
246 
247  /**
248  * Método constructor de la clase
249  * @param {string} view Nombre de la vista que genera el error
250  * @param {object} params Objeto con los parámetros a registrar
251  */
252  constructor(view = '', params = {}) {
253  // Pass remaining arguments (including vendor specific ones) to parent constructor
254  super(params);
255 
256  // Maintains proper stack trace for where our error was thrown (only available on V8)
257  if (Error.captureStackTrace) {
258  Error.captureStackTrace(this, HandleJSError);
259  }
260 
261  //console.log(params)
262  this.log_url = window.log_url || this.params.log_url || false;
263  this.debug = window.debug || this.params.debug || false;
264  this.view = view || '';
265  this.message = params.message;
266  this.date = new Date();
267  this.dataLog = {
268  view: this.view,
269  line: this.getLineNumber() || '',
270  message: this.message,
271  stacktrace: this.stack.split("\n"),
272  date: this.date
273  };
274  }
275 
276  /**
277  * Registra el evento del error generado
278  */
279  register() {
280  if (this.log_url) {
281  axios.post(this.log_url, this.dataLog).catch(error => {
282  if (this.debug) {
283  console.log(error);
284  }
285  else {
286  console.info("Error inesperado del sistema, contacte al administrador.");
287  }
288  });
289  }
290  else if (this.debug) {
291  console.log(
292  `Se generó un evento de error que no pudo ser registrado con el siguiente detalle:
293  ${this.stack}`
294  );
295  }
296  }
297 
298  /**
299  * Muestra un mensaje en la consola del navegador según el tipo indicado
300  *
301  * @param {string} type Define el tipo de error a mostrar
302  */
303  error_type(type = '') {
304  let err = this.message + "\n" + this.stack;
305  if (type === "log") {
306  console.log(err);
307  }
308  else if (type === "info") {
309  console.info(err);
310  }
311  else if (type === "warning") {
312  console.warn(err);
313  }
314  else if (type === "error") {
315  console.error(err);
316  }
317  }
