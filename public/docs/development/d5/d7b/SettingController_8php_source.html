164  {
165  /** @var array Arreglo con los parámetros a registrar */
166  $parameters = ['support', 'chat', 'notify', 'report_banner', 'multi_institution', 'digital_sign', 'online'];
167  /** @var array Arreglo con los tipos de mensaje a mostrar */
168  $msgType = ['type' => 'store'];
169 
170  foreach ($parameters as $parameter) {
171  $parameterRepository->updateOrCreate(
172  ['p_key' => $parameter, 'required_by' => 'core'],
173  ['p_value' => (!is_null($request->$parameter)) ? 'true' : 'false']
174  );
175 
176  if ($parameter === "online") {
177  if (is_null($request->$parameter)) {
178  \Artisan::call('up');
179  $msgType = [
180  'type' => 'other',
181  'text' => 'El sistema esta actualmente en línea, ' .
182  'todos los usuarios pueden acceder a la aplicación'
183  ];
184  /** @var string Título del mensaje */
185  $title = config('app.name') . ' - ' . __('En línea');
186  /** @var string Descripión del mensaje */
187  $description = __('Se ha reestablecido el acceso a la aplicación. Puede acceder en cualquier momento');
188  /** @var User Objeto con información de los usuarios del sistema */
189  $users = User::all();
190  foreach ($users as $user) {
191  $user->notify(new AppNotification($title, '', $description, true));
192  }
193  } else {
194  /** @var string Hash generado para acceder a la aplicación después de colocarla en mantenimiento */
195  $secretHash = generate_hash(36, false, true);

197  $msgType = [
198  'type' => 'other',

201  ];
202  /** @var Role Objeto con información del rol administrador */
203  $roleAdmin = Role::where('slug', 'admin')->first();
204  if ($roleAdmin && !$roleAdmin->users->isEmpty()) {
205  /** @var string Título del mensaje */
206  $title = config('app.name') . ' - ' . __('Configuración en modo mantenimiento');
207  /** @var string Descripción del mensaje */
208  $description = __('Se ha configurado la aplicación en modo mantenimiento. Para poder acceder bajo esta modalidad es necesario que acceda a la siguiente URL: ') . config('app.url') . '/' . $secretHash;
209  foreach ($roleAdmin->users as $user) {

211  }
212  }
213  }
214  }
