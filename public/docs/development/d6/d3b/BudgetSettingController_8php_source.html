95  * Guarda información de las configuraciones de presupuesto

97  * @author Ing. Roldan Vargas <rvargas@cenditel.gob.ve> | <roldandvg@gmail.com>

99  * @return Renderable
100  */
101  public function store(Request $request)
102  {
103  /** Reglas de validación para la configuración de códigos */
104  $request->validate([
105  'formulations_code' => [new CodeSettingRule],
106  'commitments_code' => [new CodeSettingRule],
107  'caused_code' => [new CodeSettingRule],
108  'payed_code' => [new CodeSettingRule],
109  'transfers_code' => [new CodeSettingRule],
110  'reductions_code' => [new CodeSettingRule],
111  'credits_code' => [new CodeSettingRule],
112  ]);
113 
114  /** @var array Arreglo con información de los campos de códigos configurados */
115  $codes = $request->input();
116  /** @var boolean Define el estatus verdadero para indicar que no se ha registrado información */
117  $saved = false;
118 
119  foreach ($codes as $key => $value) {
120  /** @var string Define el modelo al cual hace referencia el código */
121  $model = '';
122 
123  if ($key !== '_token' && !is_null($value)) {
124  list($table, $field) = explode("_", $key);
125  list($prefix, $digits, $sufix) = CodeSetting::divideCode($value);
126 
127  if ($table === "formulations") {
128  /** @var string Define el modelo para asociado a la formulación de presupuesto */
129  $model = BudgetSubSpecificFormulation::class;
130  } elseif (in_array($key, ['transfers_code', 'reductions_code', 'credits_code'])) {
131  /** @var string Define el modelo asociado a las modificaciones presupuestarias */
132  $model = BudgetModification::class;
133  /** @var string Define la tabla asociada a las modificaciones presupuestarias */
134  $table = 'modifications';
135  if ($key === 'transfers_code') {
136  /** @var string Define el tipo de registro como transferencia entre presupuestos */
137  $type = 'budget.transfers';
138  } elseif ($key === 'reductions_code') {
139  /** @var string Define el tipo de registro como reducciones presupuestarias */
140  $type = 'budget.reductions';
141  } elseif ($key === 'credits_code') {
142  /** @var string Define el tipo de registro como créditos adicionales */
143  $type = 'budget.aditional-credits';
144  }
145  } elseif ($table === "commitments") {
146  $model .= "Commitment";
147  }
148 
149  CodeSetting::updateOrCreate([
150  'module' => 'budget',
151  'table' => 'budget_' . $table,
152  'field' => $field,
153  'type' => (isset($type)) ? $type : null
154  ], [
155  'format_prefix' => $prefix,
156  'format_digits' => $digits,
157  'format_year' => $sufix,
158  'model' => $model,
159  ]);
160  /** @var boolean Define el estatus verdadero para indicar que se ha registrado información */
161  $saved = true;
162  }
163  }
164 
165  if ($saved) {
166  $request->session()->flash('message', ['type' => 'store']);
167  }
168 
169  return redirect()->back();
170  }
